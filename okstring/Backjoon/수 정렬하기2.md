### 수 정렬하기2



```swift
var cnt = Int(readLine()!)!
var arr = Array(repeating: Int.min, count: 2_000_002)

for _ in 0..<cnt {
    let num = Int(readLine()!)!
    let key = num < 0 ? 1_000_000 + num : num + 1_000_000
    arr[key] = num
}

for i in 0..<2_000_002 {
    if arr[i] != Int.min {
        print(arr[i])
    }
}
```

```swift
let n = Int(readLine()!)!
var arr = [Int]()
for _ in 0..<n {
	arr.append(Int(readLine()!)!)
}
var result = ""
for i in arr.sorted() {
	result += "\(i)\n"
}
print(result)
```
