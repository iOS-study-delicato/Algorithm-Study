let input = readLine()!.split(separator: " ").map({ Int(String($0))! })

var cNum = 0
var result = 0
loop: while true {
    cNum += 1
    for _ in 0..<cNum {
        step += 1
        if step >= start {
            result += cNum
            if step == end {
                break loop
            }
        }
    }
}
print(result)
