func solution(_ s:String) -> Bool {
    var stack = [String]()
    
    for c in Array(s) {
        if c == "(" {
            stack.append("(")
        } else if c == ")" {
            if stack.popLast() == nil {
                return false
            }
        }
    }
    return stack.isEmpty ? true : false
}
