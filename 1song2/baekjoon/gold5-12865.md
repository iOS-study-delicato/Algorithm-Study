```swift
let nk = readLine()!.split(separator: " ").map { Int(String($0))! }
let (n, k) = (nk.first!, nk.last!)
var w = [Int](repeating: 0, count: n + 1)
var v = [Int](repeating: 0, count: n + 1)
var dp = [[Int]](repeating: [Int](repeating: 0, count: k + 1), count: n + 1)

for i in 1...n {
    let wv = readLine()!.split(separator: " ").map { Int(String($0))! }
    w[i] = wv.first!
    v[i] = wv.last!
}

for i in 0..<n + 1 {
    for j in 0..<k + 1 {
        if i == 0 || j == 0 {
            dp[i][j] = 0
            continue
        }
        if w[i] <= j {
            dp[i][j] = max(dp[i - 1][j], v[i] + dp[i - 1][j - w[i]])
        } else {
            dp[i][j] = dp[i - 1][j]
        }
    }
}

print(dp[n][k])
```

let nk = readLine()!.split(separator: " ").map { Int(String($0))! }
let (n, k) = (nk.first!, nk.last!)

var items = [(Int, Int)]()
for _ in 0..<n {
    let wv = readLine()!.split(separator: " ").map { Int(String($0))! }
    if wv.first! <= k {
        items.append((wv.first!, wv.last!))
    }
}

var dp = [Int](repeating: 0, count: k + 1)
for (w, v) in items {
    var tmp = dp
    for i in w...k {
        if dp[i] < dp[i - w] + v {
            tmp[i] = dp[i - w] + v
        }
    }
    dp = tmp
}

print(dp[k])
